## Example

`my-schema.json`

    {
      "$ref": "http://json-schema.org/draft-04/schema#",
      "type": "object",
      "properties": {
        "email": {
          "type": "string",
          "format": "email"
        },
        "id": {
          "type": "string",
          "pattern": "[A-Z]{11}"
        }
      }
    }

`main.go`

    package main

    import (
        "fmt"
        "gopkg.in/matchmove/valid"
    )

    func main() {
        sampleJson := `{
          "$ref": "http://json-schema.org/draft-04/schema#",
          "type": "object",
          "properties": {
            "email": {
              "type": "string",
              "format": "email"
            },
            "id": {
              "type": "string",
              "pattern": "[A-Z]{11}"
            }
          }
        }`

        r := valid.NewJSONSchema(sampleJson).Compare(`{"email":"someone@email.com"}`)
        fmt.Print(r) //

        v := valid.NewJSONSchemaFromFile("my-schema.json").Compare(`{"email":"invalid"}`)
        fmt.Print(v.Message) // email: Does not match format 'email'
    }
